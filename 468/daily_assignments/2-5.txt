class Rabbit {
    constructor(type) {
        this.type = type;
    }

    speak(line) {
        console.log(`The ${this.type} rabbit says
        '${line}'`);
    }
}

Rabbit.prototype.speak = function(line) {
    console.log(`The rabbit of type ${this.type} says '${line}'`);
};

let killerRabbit = new Rabbit("killer");
killerRabbit.speak("Hello World!");
